// takes both a string and a character as input, and returns 
// a new string that has the passed character removed in all its positions.
const filterString  = (str, sym) => {
let i = 0;
let result = '';

while (i < str.length) {
  if (str[i] !== sym) {
    result = `${result}${str[i]}`;
    i += 1;
  }
 
  if (str[i] === sym) {
    i += 1;
  }
}
  return result;
};
